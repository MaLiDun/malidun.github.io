import numpy as np
import matplotlib.pyplot as plt
matplotlib.rcParams['axes.unicode_minus'] = False
x = [-1,1,3,5]
for i in range(1, 110):
    x.append(i/10-1)
y = [i**3+i +5 for i in x]
z = [np.sin(i) for i in y]
plt.xlim(0, 10)
plt.plot(x,z)
plt.show() # show figure
